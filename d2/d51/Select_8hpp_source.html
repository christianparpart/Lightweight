152template <typename... MoreFields>
153SqlSelectQueryBuilder& SqlSelectQueryBuilder::Fields(std::string_view const& firstField, MoreFields&&... moreFields)
154{
155 using namespace std::string_view_literals;
156
157 std::ostringstream fragment;
158
159 if (!m_query.fields.empty())
160 fragment << ", "sv;
161
162 fragment << '"' << firstField << '"';
163
164 if constexpr (sizeof...(MoreFields) > 0)
165 ((fragment << R"(, ")"sv << std::forward<MoreFields>(moreFields) << '"') << ...);
166
167 m_query.fields += fragment.str();
168 return *this;
169}
