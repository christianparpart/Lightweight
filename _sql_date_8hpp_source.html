11// Helper struct to store a date (without time of the day) to write to or read from a database.
12struct LIGHTWEIGHT_API SqlDate
13{
14 SQL_DATE_STRUCT sqlValue {};
15
16 constexpr SqlDate() noexcept = default;
17 constexpr SqlDate(SqlDate&&) noexcept = default;
18 constexpr SqlDate& operator=(SqlDate&&) noexcept = default;
19 constexpr SqlDate(SqlDate const&) noexcept = default;
20 constexpr SqlDate& operator=(SqlDate const&) noexcept = default;
21 constexpr ~SqlDate() noexcept = default;
22
23 [[nodiscard]] LIGHTWEIGHT_FORCE_INLINE constexpr std::chrono::year_month_day value() const noexcept
24 {
25 return ConvertToNative(sqlValue);
26 }
27
28 LIGHTWEIGHT_FORCE_INLINE constexpr bool operator==(SqlDate const& other) const noexcept
29 {
30 return sqlValue.year == other.sqlValue.year && sqlValue.month == other.sqlValue.month
31 && sqlValue.day == other.sqlValue.day;
32 }
33
34 LIGHTWEIGHT_FORCE_INLINE constexpr bool operator!=(SqlDate const& other) const noexcept
35 {
36 return !(*this == other);
37 }
38
39 constexpr SqlDate(std::chrono::year_month_day value) noexcept:
40 sqlValue { SqlDate::ConvertToSqlValue(value) }
41 {
42 }
43
44 constexpr SqlDate(std::chrono::year year, std::chrono::month month, std::chrono::day day) noexcept:
45 SqlDate(std::chrono::year_month_day { year, month, day })
46 {
47 }
48
49 static LIGHTWEIGHT_FORCE_INLINE SqlDate Today() noexcept
50 {
51 return SqlDate { std::chrono::year_month_day {
52 std::chrono::floor<std::chrono::days>(std::chrono::system_clock::now()),
53 } };
54 }
55
56 static LIGHTWEIGHT_FORCE_INLINE constexpr SQL_DATE_STRUCT ConvertToSqlValue(
57 std::chrono::year_month_day value) noexcept
58 {
59 return SQL_DATE_STRUCT {
60 .year = (SQLSMALLINT) (int) value.year(),
61 .month = (SQLUSMALLINT) (unsigned) value.month(),
62 .day = (SQLUSMALLINT) (unsigned) value.day(),
63 };
64 }
65
66 static LIGHTWEIGHT_FORCE_INLINE constexpr std::chrono::year_month_day ConvertToNative(
67 SQL_DATE_STRUCT const& value) noexcept
68 {
69 return std::chrono::year_month_day { std::chrono::year { value.year },
70 std::chrono::month { static_cast<unsigned>(value.month) },
71 std::chrono::day { static_cast<unsigned>(value.day) } };
72 }
73};
74
75template <>
76struct std::formatter<SqlDate>: std::formatter<std::string>
77{
78 auto format(SqlDate const& value, std::format_context& ctx) const -> std::format_context::iterator
79 {
80 return std::formatter<std::string>::format(
81 std::format("{:04}-{:02}-{:02}", value.sqlValue.year, value.sqlValue.month, value.sqlValue.day), ctx);
82 }
83};

112
113 static LIGHTWEIGHT_FORCE_INLINE SQLRETURN GetColumn(SQLHSTMT stmt,
114 SQLUSMALLINT column,

118 {

120 }

122 static LIGHTWEIGHT_FORCE_INLINE std::string Inspect(SqlDate const& value) noexcept
