351/// SQL query result row iterator
352template <typename T>
353class SqlRowIterator
354{
355 public:
356 using difference_type = bool;
357 using value_type = T;
358
359 SqlRowIterator(SqlStatement& stmt):
360 _stmt(&stmt)
361 {
362 }
363
364 SqlRowIterator operator++()
365 {
366 _is_end = _stmt->FetchRow();
367 return *this;
368 }
369
370 value_type operator*()
371 {
372 auto res = T {};
373
374 Reflection::EnumerateMembers(res, [this]<size_t I>(auto&& value) {
375 auto tmp = _stmt->GetColumn<typename Reflection::MemberTypeOf<I, value_type>::ValueType>(I + 1);
376 value = tmp;
377 });
378
379 return res;
380 }
381
382 SqlRowIterator begin()
383 {
384 auto query =
385 _stmt->Query(detail::RecordTableName<value_type>::Value).Select().template Fields<value_type>().All();
386 _stmt->Prepare(query);
387 _stmt->Execute();
388 _is_end = _stmt->FetchRow();
389 return *this;
390 }
391
392 SqlRowIterator end()
393 {
394 auto end_iterator = SqlRowIterator<T>(*_stmt);
395 end_iterator._is_end = true;
396 return end_iterator;
397 }
398
399 bool operator!=(SqlRowIterator const& other) const
400 {
401 return _is_end != other._is_end;
402 }
403
404 private:
405 SqlStatement* _stmt;
406 bool _is_end = false;
407};

506template <SqlInputParameterBinder... Args>
507void SqlStatement::Execute(Args const&... args)
508{
509 // Each input parameter must have an address,
510 // such that we can call SQLBindParameter() without needing to copy it.
511 // The memory region behind the input parameter must exist until the SQLExecute() call.
512
513 SqlLogger::GetLogger().OnExecute(m_preparedQuery);
514
515 if (!(m_expectedParameterCount == (std::numeric_limits<decltype(m_expectedParameterCount)>::max)()
516 && sizeof...(args) == 0)
517 && !(m_expectedParameterCount == sizeof...(args)))
518 throw std::invalid_argument { "Invalid argument count" };
519
520 SQLUSMALLINT i = 0;
521 ((++i,
522 SqlLogger::GetLogger().OnBindInputParameter({}, args),
523 RequireSuccess(SqlDataBinder<Args>::InputParameter(m_hStmt, i, args, *this))),
524 ...);
525
526 auto const result = SQLExecute(m_hStmt);
527
528 if (result != SQL_NO_DATA && result != SQL_SUCCESS && result != SQL_SUCCESS_WITH_INFO)
529 throw SqlException(SqlErrorInfo::fromStatementHandle(m_hStmt), std::source_location::current());
530
531 ProcessPostExecuteCallbacks();
532}

554template <typename FirstColumnBatch, typename... MoreColumnBatches>
555concept SqlNativeBatchable =
556 std::ranges::contiguous_range<FirstColumnBatch>
557 && (std::ranges::contiguous_range<MoreColumnBatches> && ...)
558 && SqlNativeContiguousValueConcept<std::ranges::range_value_t<FirstColumnBatch>>
559 && (SqlNativeContiguousValueConcept<std::ranges::range_value_t<MoreColumnBatches>> && ...);

561// clang-format on
562
563template <SqlInputParameterBatchBinder FirstColumnBatch, std::ranges::contiguous_range... MoreColumnBatches>
564void SqlStatement::ExecuteBatchNative(FirstColumnBatch const& firstColumnBatch,
565 MoreColumnBatches const&... moreColumnBatches)
566{
567 static_assert(SqlNativeBatchable<FirstColumnBatch, MoreColumnBatches...>,
568 "Must be a supported native contiguous element type.");
569
570 if (m_expectedParameterCount != 1 + sizeof...(moreColumnBatches))
571 throw std::invalid_argument { "Invalid number of columns" };
572
573 const auto rowCount = std::ranges::size(firstColumnBatch);
574 if (!((std::size(moreColumnBatches) == rowCount) && ...))
575 throw std::invalid_argument { "Uneven number of rows" };
576
577 size_t rowStart = 0;
578
579 // clang-format off
580 // NOLINTNEXTLINE(performance-no-int-to-ptr)
581 RequireSuccess(SQLSetStmtAttr(m_hStmt, SQL_ATTR_PARAMSET_SIZE, (SQLPOINTER) rowCount, 0));
582 RequireSuccess(SQLSetStmtAttr(m_hStmt, SQL_ATTR_PARAM_BIND_OFFSET_PTR, &rowStart, 0));
583 RequireSuccess(SQLSetStmtAttr(m_hStmt, SQL_ATTR_PARAM_BIND_TYPE, SQL_PARAM_BIND_BY_COLUMN, 0));
584 RequireSuccess(SQLSetStmtAttr(m_hStmt, SQL_ATTR_PARAM_OPERATION_PTR, SQL_PARAM_PROCEED, 0));
585 RequireSuccess(SqlDataBinder<std::remove_cvref_t<decltype(*std::ranges::data(firstColumnBatch))>>::
586 InputParameter(m_hStmt, 1, *std::ranges::data(firstColumnBatch), *this));
587 SQLUSMALLINT column = 1;
588 (RequireSuccess(SqlDataBinder<std::remove_cvref_t<decltype(*std::ranges::data(moreColumnBatches))>>::
589 InputParameter(m_hStmt, ++column, *std::ranges::data(moreColumnBatches), *this)), ...);
590 RequireSuccess(SQLExecute(m_hStmt));
591 ProcessPostExecuteCallbacks();
592 // clang-format on
593}

595template <SqlInputParameterBatchBinder FirstColumnBatch, std::ranges::range... MoreColumnBatches>
596inline LIGHTWEIGHT_FORCE_INLINE void SqlStatement::ExecuteBatch(FirstColumnBatch const& firstColumnBatch,
597 MoreColumnBatches const&... moreColumnBatches)
598{
599 // If the input ranges are contiguous and their element types are contiguous and supported as well,
600 // we can use the native batch execution.
601 if constexpr (SqlNativeBatchable<FirstColumnBatch, MoreColumnBatches...>)
602 ExecuteBatchNative(firstColumnBatch, moreColumnBatches...);
603 else
604 ExecuteBatchSoft(firstColumnBatch, moreColumnBatches...);
605}
606
607template <SqlInputParameterBatchBinder FirstColumnBatch, std::ranges::range... MoreColumnBatches>
608void SqlStatement::ExecuteBatchSoft(FirstColumnBatch const& firstColumnBatch,
609 MoreColumnBatches const&... moreColumnBatches)
610{
611 if (m_expectedParameterCount != 1 + sizeof...(moreColumnBatches))
612 throw std::invalid_argument { "Invalid number of columns" };
613
614 const auto rowCount = std::ranges::size(firstColumnBatch);
615 if (!((std::size(moreColumnBatches) == rowCount) && ...))
616 throw std::invalid_argument { "Uneven number of rows" };
617
618 for (auto const rowIndex: std::views::iota(size_t { 0 }, rowCount))
619 {
620 std::apply(
621 [&]<SqlInputParameterBinder... ColumnValues>(ColumnValues const&... columnsInRow) {
622 SQLUSMALLINT column = 0;
623 ((++column, SqlDataBinder<ColumnValues>::InputParameter(m_hStmt, column, columnsInRow, *this)), ...);
624 RequireSuccess(SQLExecute(m_hStmt));
625 ProcessPostExecuteCallbacks();
626 },
627 std::make_tuple(std::ref(*std::ranges::next(std::ranges::begin(firstColumnBatch), rowIndex)),
628 std::ref(*std::ranges::next(std::ranges::begin(moreColumnBatches), rowIndex))...));
629 }
630}

632template <SqlGetColumnNativeType T>
633inline bool SqlStatement::GetColumn(SQLUSMALLINT column, T* result) const
634{
635 SQLLEN indicator {}; // TODO: Handle NULL values if we find out that we need them for our use-cases.
