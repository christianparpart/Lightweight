342// input iterator
343template <typename T>
344class SqlRowIterator
345{
346 public:
347 using difference_type = bool;
348 using value_type = T;
349
350 SqlRowIterator(SqlStatement& stmt):
351 _stmt(&stmt)
352 {
353 }
354
355 SqlRowIterator operator++()
356 {
357 _is_end = _stmt->FetchRow();
358 return *this;
359 }
360
361 value_type operator*()
362 {
363 auto res = T {};
364
365 Reflection::EnumerateMembers(res, [this]<size_t I>(auto&& value) {
366 auto tmp = _stmt->GetColumn<typename Reflection::MemberTypeOf<I, value_type>::ValueType>(I + 1);
367 value = tmp;
368 });
369
370 return res;
371 }
372
373 SqlRowIterator begin()
374 {
375 auto query =
376 _stmt->Query(detail::RecordTableName<value_type>::Value).Select().template Fields<value_type>().All();
377 _stmt->Prepare(query);
378 _stmt->Execute();
379 _is_end = _stmt->FetchRow();
380 return *this;
381 }
382
383 SqlRowIterator end()
384 {
385 auto end_iterator = SqlRowIterator<T>(*_stmt);
386 end_iterator._is_end = true;
387 return end_iterator;
388 }
389
390 bool operator!=(SqlRowIterator const& other) const
391 {
392 return _is_end != other._is_end;
393 }
394
395 private:
396 SqlStatement* _stmt;
397 bool _is_end = false;
398};

481template <SqlInputParameterBinder... Args>
482void SqlStatement::Execute(Args const&... args)
483{
484 // Each input parameter must have an address,
485 // such that we can call SQLBindParameter() without needing to copy it.
486 // The memory region behind the input parameter must exist until the SQLExecute() call.
487
488 SqlLogger::GetLogger().OnExecute(m_preparedQuery);
489
490 if (!(m_expectedParameterCount == (std::numeric_limits<decltype(m_expectedParameterCount)>::max)()
491 && sizeof...(args) == 0)
492 && !(m_expectedParameterCount == sizeof...(args)))
493 throw std::invalid_argument { "Invalid argument count" };
494
495 SQLUSMALLINT i = 0;
496 ((++i,
497 SqlLogger::GetLogger().OnBindInputParameter({}, args),
498 RequireSuccess(SqlDataBinder<Args>::InputParameter(m_hStmt, i, args, *this))),
499 ...);
500
501 auto const result = SQLExecute(m_hStmt);
502
503 if (result != SQL_NO_DATA && result != SQL_SUCCESS && result != SQL_SUCCESS_WITH_INFO)
504 throw SqlException(SqlErrorInfo::fromStatementHandle(m_hStmt), std::source_location::current());
505
506 ProcessPostExecuteCallbacks();
507}

529template <typename FirstColumnBatch, typename... MoreColumnBatches>
530concept SqlNativeBatchable =
531 std::ranges::contiguous_range<FirstColumnBatch>
532 && (std::ranges::contiguous_range<MoreColumnBatches> && ...)
533 && SqlNativeContiguousValueConcept<std::ranges::range_value_t<FirstColumnBatch>>
534 && (SqlNativeContiguousValueConcept<std::ranges::range_value_t<MoreColumnBatches>> && ...);
535
536// clang-format on
537
538template <SqlInputParameterBatchBinder FirstColumnBatch, std::ranges::contiguous_range... MoreColumnBatches>
539void SqlStatement::ExecuteBatchNative(FirstColumnBatch const& firstColumnBatch,
540 MoreColumnBatches const&... moreColumnBatches)
541{
542 static_assert(SqlNativeBatchable<FirstColumnBatch, MoreColumnBatches...>,
543 "Must be a supported native contiguous element type.");
544
545 if (m_expectedParameterCount != 1 + sizeof...(moreColumnBatches))
546 throw std::invalid_argument { "Invalid number of columns" };
547
548 const auto rowCount = std::ranges::size(firstColumnBatch);
549 if (!((std::size(moreColumnBatches) == rowCount) && ...))
550 throw std::invalid_argument { "Uneven number of rows" };
551
552 size_t rowStart = 0;
553
554 // clang-format off
555 // NOLINTNEXTLINE(performance-no-int-to-ptr)
556 RequireSuccess(SQLSetStmtAttr(m_hStmt, SQL_ATTR_PARAMSET_SIZE, (SQLPOINTER) rowCount, 0));
557 RequireSuccess(SQLSetStmtAttr(m_hStmt, SQL_ATTR_PARAM_BIND_OFFSET_PTR, &rowStart, 0));
558 RequireSuccess(SQLSetStmtAttr(m_hStmt, SQL_ATTR_PARAM_BIND_TYPE, SQL_PARAM_BIND_BY_COLUMN, 0));
559 RequireSuccess(SQLSetStmtAttr(m_hStmt, SQL_ATTR_PARAM_OPERATION_PTR, SQL_PARAM_PROCEED, 0));
560 RequireSuccess(SqlDataBinder<std::remove_cvref_t<decltype(*std::ranges::data(firstColumnBatch))>>::
561 InputParameter(m_hStmt, 1, *std::ranges::data(firstColumnBatch), *this));
562 SQLUSMALLINT column = 1;
563 (RequireSuccess(SqlDataBinder<std::remove_cvref_t<decltype(*std::ranges::data(moreColumnBatches))>>::
564 InputParameter(m_hStmt, ++column, *std::ranges::data(moreColumnBatches), *this)), ...);
565 RequireSuccess(SQLExecute(m_hStmt));
566 ProcessPostExecuteCallbacks();
567 // clang-format on
568}

570template <SqlInputParameterBatchBinder FirstColumnBatch, std::ranges::range... MoreColumnBatches>
571inline LIGHTWEIGHT_FORCE_INLINE void SqlStatement::ExecuteBatch(FirstColumnBatch const& firstColumnBatch,
572 MoreColumnBatches const&... moreColumnBatches)
573{
574 // If the input ranges are contiguous and their element types are contiguous and supported as well,
575 // we can use the native batch execution.
576 if constexpr (SqlNativeBatchable<FirstColumnBatch, MoreColumnBatches...>)
577 ExecuteBatchNative(firstColumnBatch, moreColumnBatches...);
578 else
579 ExecuteBatchSoft(firstColumnBatch, moreColumnBatches...);
580}
581
582template <SqlInputParameterBatchBinder FirstColumnBatch, std::ranges::range... MoreColumnBatches>
583void SqlStatement::ExecuteBatchSoft(FirstColumnBatch const& firstColumnBatch,
584 MoreColumnBatches const&... moreColumnBatches)
585{
586 if (m_expectedParameterCount != 1 + sizeof...(moreColumnBatches))
587 throw std::invalid_argument { "Invalid number of columns" };
588
589 const auto rowCount = std::ranges::size(firstColumnBatch);
590 if (!((std::size(moreColumnBatches) == rowCount) && ...))
591 throw std::invalid_argument { "Uneven number of rows" };
592
593 for (auto const rowIndex: std::views::iota(size_t { 0 }, rowCount))
594 {
595 std::apply(
596 [&]<SqlInputParameterBinder... ColumnValues>(ColumnValues const&... columnsInRow) {
597 SQLUSMALLINT column = 0;
598 ((++column, SqlDataBinder<ColumnValues>::InputParameter(m_hStmt, column, columnsInRow, *this)), ...);
599 RequireSuccess(SQLExecute(m_hStmt));
600 ProcessPostExecuteCallbacks();
601 },
602 std::make_tuple(std::ref(*std::ranges::next(std::ranges::begin(firstColumnBatch), rowIndex)),
603 std::ref(*std::ranges::next(std::ranges::begin(moreColumnBatches), rowIndex))...));
604 }
605}
